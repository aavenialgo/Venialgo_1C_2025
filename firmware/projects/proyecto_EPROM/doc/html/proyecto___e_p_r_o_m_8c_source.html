84static void processAndSendEcg(void *pvParameter)
85{
86 uint16_t valor = 0;
87 char msg[128];
88 char msg_chunk[24];
89 static uint8_t i = 0;
90 while (1)
91 {
92 ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
93 AnalogInputReadSingle(CH2, &valor);
94 ecg_muestra[i] = valor;
95 valor_actual = valor;
96 i++;
97
98 if (i == CHUNK)
99 {
100 HiPassFilter(&ecg_muestra[0], ecg_filt, CHUNK);
101 LowPassFilter(ecg_filt, ecg_filt, CHUNK);
102 strcpy(msg, "");
103
104 for (uint8_t j = 0; j < CHUNK; j++)
105 {
106 sprintf(msg_chunk, "*G%.2f*", ecg_filt[j]);
107 strcat(msg, msg_chunk);
108 }
109 BleSendString(msg);
110 i = 0;
111 }
112 }
113}
119void procesAndSendPpg(void *pvParameter)
120{ char msg[32];
121 float analog_raw, analog_hp, analog_filt;
122 uint16_t adcValue = 0;
123 while(true){
124 ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
125
126 AnalogInputReadSingle(poteInput.input, &adcValue);
127 analog_raw = (float)adcValue;
128
129 HiPassFilter(&analog_raw, &analog_hp, 1);
130 LowPassFilter(&analog_hp, &analog_filt, 1);
131 snprintf(msg, sizeof(msg), "%0.2f\n", analog_filt);
132 printf(">PPG:%0.2f\r\n", analog_filt);
133 }
134}
