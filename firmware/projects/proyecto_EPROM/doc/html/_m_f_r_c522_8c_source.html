160void PCD_ReadNRegister(
161 MFRC522Ptr_t mfrc,
162 uint8_t reg,
163 uint8_t count,
164 uint8_t *values,
165 uint8_t rxAlign
166 ) {
167 // rxAlign=0;
168 if (count == 0) {
169 return;
170 }
171 // UartSendString(SERIAL_PORT_PC,F("Reading ")); UartSendString(SERIAL_PORT_PC,count); UartSendString(SERIAL_PORT_PC,F(" uint8_ts from
172 // register."));
173 uint8_t address = 0x80 | (reg & 0x7E); // MSB == 1 is for reading. LSB is
174 // not used in address. Datasheet
175 // section 8.1.2.3.
176 uint8_t index = 0; // Index in values array.
177
178 // Set the settings to work with SPI bus
179 SpiInit(&spi_conf);
180
181 // Select slave
182 GPIOOff(mfrc522_dc);
183
184 // One read is performed outside of the loop
185 count--;
186
187 // Tell MFRC522 which address we want to read
188 // SPI.transfer(address);
189 mfrc->Tx_Buf[0] = address;
190 SpiWrite(mfrc522_spi, mfrc->Tx_Buf, 1);
191
192
193 while (index < count) {
194 if (index == 0 &&
195 rxAlign) { // Only update bit positions rxAlign..7 in values[0]
196 // Create bit mask for bit positions rxAlign..7
197 uint8_t mask = 0;
198 uint8_t i;
199 for (i = rxAlign; i <= 7; i++) {
200 mask |= (1 << i);
201 }
202
203 // Read value and tell that we want to read the same address again.
204 // uint8_t value = SPI.transfer(address);
205 mfrc->Tx_Buf[0] = address;
206 SpiReadWrite(mfrc522_spi, mfrc->Tx_Buf, mfrc->Rx_Buf, 1);
207 uint8_t value = mfrc->Rx_Buf[0];
208
209 // Apply mask to both current value of values[0] and the new data in
210 // value.
211 values[0] = (values[index] & ~mask) | (value & mask);
212 } else { // Normal case
213 // Read value and tell that we want to read the same address again.
214 // values[index] = SPI.transfer(address);
215 mfrc->Tx_Buf[0] = address;
216 SpiReadWrite(mfrc522_spi, mfrc->Tx_Buf, mfrc->Rx_Buf, 1);
217 values[index] = mfrc->Rx_Buf[0];
218 }
219 index++;
220 }
221
222 // Read the final uint8_t. Send 0 to stop reading.
223 // values[index] = SPI.transfer(0);
224 mfrc->Tx_Buf[0] = 0;
225 SpiReadWrite(mfrc522_spi, mfrc->Tx_Buf, mfrc->Rx_Buf, 1);
226 values[index] = mfrc->Rx_Buf[0];
227
228 // Release slave again
229 GPIOOn(mfrc522_dc);
230} // End PCD_ReadRegister()

263StatusCode
264PCD_CalculateCRC(MFRC522Ptr_t mfrc,
265 uint8_t *data,
267 uint8_t length,
268 uint8_t *result
270 ) {
271 PCD_WriteRegister(mfrc, CommandReg, PCD_Idle); // Stop any active command.
272 PCD_WriteRegister(mfrc, DivIrqReg,
273 0x04); // Clear the CRCIRq interrupt request bit
274 PCD_SetRegisterBitMask(mfrc, FIFOLevelReg,
275 0x80); // FlushBuffer = 1, FIFO initialization
276 PCD_WriteNRegister(mfrc, FIFODataReg, length,
277 data); // Write data to the FIFO
278 PCD_WriteRegister(mfrc, CommandReg, PCD_CalcCRC); // Start the calculation
279
280 // Wait for the CRC calculation to complete. Each iteration of the
281 // while-loop takes 17.73�s.
282 uint16_t i = 5000;
283 uint8_t n;
284 while (1) {
285 n = PCD_ReadRegister(mfrc, DivIrqReg); // DivIrqReg[7..0] bits are: Set2
286 // reserved reserved MfinActIRq
287 // reserved CRCIRq reserved
288 // reserved
289 if (n & 0x04) { // CRCIRq bit set - calculation done
290 break;
291 }
292 if (--i == 0) { // The emergency break. We will eventually terminate on
293 // this one after 89ms. Communication with the MFRC522
294 // might be down.
295 return STATUS_TIMEOUT;
296 }
297 }
298 PCD_WriteRegister(
299 mfrc, CommandReg,
300 PCD_Idle); // Stop calculating CRC for new content in the FIFO.
301
302 // Transfer the result from the registers to the result buffer
303 result[0] = PCD_ReadRegister(mfrc, CRCResultRegL);
304 result[1] = PCD_ReadRegister(mfrc, CRCResultRegH);
305 return STATUS_OK;
306} // End PCD_CalculateCRC()

315void PCD_Init(MFRC522Ptr_t mfrc) {
316
317 /* SPI configuration */
318 spi_conf.device = mfrc->spi_dev;
319 mfrc522_spi = mfrc->spi_dev;
320 /* GPIOs configuration and initialization */
321 mfrc522_dc = mfrc->_chipSelectPin;
322 mfrc522_rst = mfrc->_resetPowerDownPin;
323 GPIOInit(mfrc522_dc, GPIO_OUTPUT);
324 GPIOInit(mfrc522_rst, GPIO_OUTPUT);
325
326 DelayUs(10);
327 GPIOOn(mfrc522_dc);
328
329 if (GPIORead(mfrc522_rst) == false) { // The MFRC522 chip is in power down mode.
330 GPIOOn(mfrc522_rst); // Exit power down mode. This triggers a hard reset.
331 // Section 8.8.2 in the datasheet says the oscillator start-up time is
332 // the start up time of the crystal + 37,74�s. Let us be generous: 50ms.
333 //SysTick_Init();
334 //delay_ms(50);
335 //Board_LED_Toggle(0);
336 } else { // Perform a soft reset
337 PCD_Reset(mfrc);
338 }
339
340 // When communicating with a PICC we need a timeout if something goes wrong.
341 // f_timer = 13.56 MHz / (2*TPreScaler+1) where TPreScaler =
342 // [TPrescaler_Hi:TPrescaler_Lo].
343 // TPrescaler_Hi are the four low bits in TModeReg. TPrescaler_Lo is
344 // TPrescalerReg.
345 PCD_WriteRegister(mfrc, TModeReg, 0x80); // TAuto=1; timer starts
346 // automatically at the end of the
347 // transmission in all
348 // communication modes at all
349 // speeds
350 PCD_WriteRegister(mfrc, TPrescalerReg,
351 0xA9); // TPreScaler = TModeReg[3..0]:TPrescalerReg, ie
352 // 0x0A9 = 169 => f_timer=40kHz, ie a timer period
353 // of 25�s.
354 PCD_WriteRegister(
355 mfrc, TReloadRegH,
356 0x03); // Reload timer with 0x3E8 = 1000, ie 25ms before timeout.
357 PCD_WriteRegister(mfrc, TReloadRegL, 0xE8);
358
359 PCD_WriteRegister(mfrc, TxASKReg, 0x40); // Default 0x00. Force a 100 % ASK
360 // modulation independent of the
361 // ModGsPReg register setting
362 PCD_WriteRegister(mfrc, ModeReg, 0x3D); // Default 0x3F. Set the preset
363 // value for the CRC coprocessor for
364 // the CalcCRC command to 0x6363
365 // (ISO 14443-3 part 6.2.4)
366 PCD_AntennaOn(mfrc); // Enable the antenna driver pins TX1 and TX2 (they
367 // were disabled by the reset)
368} // End PCD_Init()

478StatusCode PCD_CommunicateWithPICC(
479 MFRC522Ptr_t mfrc,
480 uint8_t command,
481 uint8_t waitIRq,
483 uint8_t *sendData,
484 uint8_t sendLen,
485 uint8_t *backData,
487 uint8_t *backLen,
489 uint8_t *validBits,
491 uint8_t rxAlign,
493 bool checkCRC
495 ) {
496 // backData=NULL;
497 // backLen=NULL;
498 // validBits=NULL;
499 // rxAlign=0;
500 // checkCRC=false;
501
502 uint8_t n, _validBits;
503 unsigned int i;
504
505 // Prepare values for BitFramingReg
506 uint8_t txLastBits = validBits ? *validBits : 0;
507 uint8_t bitFraming =
508 (rxAlign << 4) + txLastBits; // RxAlign = BitFramingReg[6..4].
509 // TxLastBits = BitFramingReg[2..0]
510
511 PCD_WriteRegister(mfrc, CommandReg, PCD_Idle); // Stop any active command.
512 PCD_WriteRegister(mfrc, ComIrqReg,
513 0x7F); // Clear all seven interrupt request bits
514 PCD_SetRegisterBitMask(mfrc, FIFOLevelReg,
515 0x80); // FlushBuffer = 1, FIFO initialization
516 PCD_WriteNRegister(mfrc, FIFODataReg, sendLen,
517 sendData); // Write sendData to the FIFO
518 PCD_WriteRegister(mfrc, BitFramingReg, bitFraming); // Bit adjustments
519 PCD_WriteRegister(mfrc, CommandReg, command); // Execute the command
520 if (command == PCD_Transceive) {
521 PCD_SetRegisterBitMask(
522 mfrc, BitFramingReg,
523 0x80); // StartSend=1, transmission of data starts
524 }
525
526 // Wait for the command to complete.
527 // In PCD_Init() we set the TAuto flag in TModeReg. This means the timer
528 // automatically starts when the PCD stops transmitting.
529 // Each iteration of the do-while-loop takes 17.86�s.
530 i = 2000;
531 while (1) {
532 n = PCD_ReadRegister(mfrc, ComIrqReg); // ComIrqReg[7..0] bits are: Set1
533 // TxIRq RxIRq IdleIRq HiAlertIRq
534 // LoAlertIRq ErrIRq TimerIRq
535 if (n & waitIRq) { // One of the interrupts that signal success has been
536 // set.
537 break;
538 }
539 if (n & 0x01) { // Timer interrupt - nothing received in 25ms
540 return STATUS_TIMEOUT;
541 }
542 if (--i == 0) { // The emergency break. If all other conditions fail we
543 // will eventually terminate on this one after 35.7ms.
544 // Communication with the MFRC522 might be down.
545 return STATUS_TIMEOUT;
546 }
547 }
548
549 // Stop now if any errors except collisions were detected.
550 uint8_t errorRegValue =
551 PCD_ReadRegister(mfrc, ErrorReg); // ErrorReg[7..0] bits are: WrErr
552 // TempErr reserved BufferOvfl CollErr
553 // CRCErr ParityErr ProtocolErr
554 if (errorRegValue & 0x13) { // BufferOvfl ParityErr ProtocolErr
555 return STATUS_ERROR;
556 }
557
558 // If the caller wants data back, get it from the MFRC522.
559 if (backData && backLen) {
560 n = PCD_ReadRegister(mfrc,
561 FIFOLevelReg); // Number of uint8_ts in the FIFO
562 if (n > *backLen) {
563 return STATUS_NO_ROOM;
564 }
565 *backLen = n; // Number of uint8_ts returned
566 PCD_ReadNRegister(mfrc, FIFODataReg, n, backData,
567 rxAlign); // Get received data from FIFO
568 _validBits = PCD_ReadRegister(mfrc, ControlReg) &
569 0x07; // RxLastBits[2:0] indicates the number of valid bits
570 // in the last received uint8_t. If this value is
571 // 000b, the whole uint8_t is valid.
572 if (validBits) {
573 *validBits = _validBits;
574 }
575 }
576
577 // Tell about collisions
578 if (errorRegValue & 0x08) { // CollErr
579 return STATUS_COLLISION;
580 }
581
582 // Perform CRC_A validation if requested.
583 if (backData && backLen && checkCRC) {
584 // In this case a MIFARE Classic NAK is not OK.
585 if (*backLen == 1 && _validBits == 4) {
586 return STATUS_MIFARE_NACK;
587 }
588 // We need at least the CRC_A value and all 8 bits of the last uint8_t
589 // must be received.
590 if (*backLen < 2 || _validBits != 0) {
591 return STATUS_CRC_WRONG;
592 }
593 // Verify CRC_A - do our own calculation and store the control in
594 // controlBuffer.
595 uint8_t controlBuffer[2];
596 StatusCode status = PCD_CalculateCRC(mfrc, &backData[0], *backLen - 2,
597 &controlBuffer[0]);
598 if (status != STATUS_OK) {
599 return status;
600 }
601 if ((backData[*backLen - 2] != controlBuffer[0]) ||
602 (backData[*backLen - 1] != controlBuffer[1])) {
603 return STATUS_CRC_WRONG;
604 }
605 }
606
607 return STATUS_OK;
608} // End PCD_CommunicateWithPICC()

710StatusCode PICC_Select(MFRC522Ptr_t mfrc,
711 Uid *uid,
713 uint8_t validBits
716 ) {
717 // validBits=0;
718
719 bool uidComplete;
720 bool selectDone;
721 bool useCascadeTag;
722 uint8_t cascadeLevel = 1;
723 StatusCode result;
724 uint8_t count;
725 uint8_t index;
726 uint8_t uidIndex; // The first index in uid->uiduint8_t[] that is used in
727 // the current Cascade Level.
728 int8_t currentLevelKnownBits; // The number of known UID bits in the current
729 // Cascade Level.
730 uint8_t buffer[9]; // The SELECT/ANTICOLLISION commands uses a 7 uint8_t
731 // standard frame + 2 uint8_ts CRC_A
732 uint8_t bufferUsed; // The number of uint8_ts used in the buffer, ie the
733 // number of uint8_ts to transfer to the FIFO.
734 uint8_t rxAlign; // Used in BitFramingReg. Defines the bit position for the
735 // first bit received.
736 uint8_t txLastBits; // Used in BitFramingReg. The number of valid bits in
737 // the last transmitted uint8_t.
738 uint8_t *responseBuffer;
739 uint8_t responseLength;
740
741 // Description of buffer structure:
742 // uint8_t 0: SEL Indicates the Cascade Level: PICC_CMD_SEL_CL1,
743 //PICC_CMD_SEL_CL2 or PICC_CMD_SEL_CL3
744 // uint8_t 1: NVB Number of Valid Bits (in complete command, not
745 //just the UID): High nibble: complete uint8_ts, Low nibble: Extra bits.
746 // uint8_t 2: UID-data or CT See explanation below. CT means Cascade
747 //Tag.
748 // uint8_t 3: UID-data
749 // uint8_t 4: UID-data
750 // uint8_t 5: UID-data
751 // uint8_t 6: BCC Block Check Character - XOR of uint8_ts
752 //2-5
753 // uint8_t 7: CRC_A
754 // uint8_t 8: CRC_A
755 // The BCC and CRC_A are only transmitted if we know all the UID bits of the
756 // current Cascade Level.
757 //
758 // Description of uint8_ts 2-5: (Section 6.5.4 of the ISO/IEC 14443-3 draft:
759 // UID contents and cascade levels)
760 // UID size Cascade level uint8_t2 uint8_t3 uint8_t4
761 //uint8_t5
762 // ======== ============= ===== ===== ===== =====
763 // 4 uint8_ts 1 uid0 uid1 uid2 uid3
764 // 7 uint8_ts 1 CT uid0 uid1 uid2
765 // 2 uid3 uid4 uid5 uid6
766 // 10 uint8_ts 1 CT uid0 uid1 uid2
767 // 2 CT uid3 uid4 uid5
768 // 3 uid6 uid7 uid8 uid9
769
770 // Sanity checks
771 if (validBits > 80) {
772 return STATUS_INVALID;
773 }
774
775 // Prepare MFRC522
776 PCD_ClearRegisterBitMask(mfrc, CollReg, 0x80); // ValuesAfterColl=1 => Bits
777 // received after collision
778 // are cleared.
779
780 // Repeat Cascade Level loop until we have a complete UID.
781 uidComplete = false;
782 while (!uidComplete) {
783 // Set the Cascade Level in the SEL uint8_t, find out if we need to use
784 // the Cascade Tag in uint8_t 2.
785 switch (cascadeLevel) {
786 case 1:
787 buffer[0] = PICC_CMD_SEL_CL1;
788 uidIndex = 0;
789 useCascadeTag =
790 validBits &&
791 uid->size >
792 4; // When we know that the UID has more than 4 uint8_ts
793 break;
794
795 case 2:
796 buffer[0] = PICC_CMD_SEL_CL2;
797 uidIndex = 3;
798 useCascadeTag =
799 validBits &&
800 uid->size >
801 7; // When we know that the UID has more than 7 uint8_ts
802 break;
803
804 case 3:
805 buffer[0] = PICC_CMD_SEL_CL3;
806 uidIndex = 6;
807 useCascadeTag = false; // Never used in CL3.
808 break;
809
810 default:
811 return STATUS_INTERNAL_ERROR;
812 break;
813 }
814
815 // How many UID bits are known in this Cascade Level?
816 currentLevelKnownBits = validBits - (8 * uidIndex);
817 if (currentLevelKnownBits < 0) {
818 currentLevelKnownBits = 0;
819 }
820 // Copy the known bits from uid->uiduint8_t[] to buffer[]
821 index = 2; // destination index in buffer[]
822 if (useCascadeTag) {
823 buffer[index++] = PICC_CMD_CT;
824 }
825 uint8_t uint8_tsToCopy =
826 currentLevelKnownBits / 8 +
827 (currentLevelKnownBits % 8 ? 1 : 0); // The number of uint8_ts
828 // needed to represent the
829 // known bits for this level.
830 if (uint8_tsToCopy) {
831 uint8_t maxuint8_ts =
832 useCascadeTag ? 3 : 4; // Max 4 uint8_ts in each Cascade Level.
833 // Only 3 left if we use the Cascade Tag
834 if (uint8_tsToCopy > maxuint8_ts) {
835 uint8_tsToCopy = maxuint8_ts;
836 }
837 for (count = 0; count < uint8_tsToCopy; count++) {
838 buffer[index++] = uid->uidByte[uidIndex + count];
839 }
840 }
841 // Now that the data has been copied we need to include the 8 bits in CT
842 // in currentLevelKnownBits
843 if (useCascadeTag) {
844 currentLevelKnownBits += 8;
845 }
846
847 // Repeat anti collision loop until we can transmit all UID bits + BCC
848 // and receive a SAK - max 32 iterations.
849 selectDone = false;
850 while (!selectDone) {
851 // Find out how many bits and uint8_ts to send and receive.
852 if (currentLevelKnownBits >= 32) { // All UID bits in this Cascade
853 // Level are known. This is a
854 // SELECT.
855 // UartSendString(SERIAL_PORT_PC,F("SELECT: currentLevelKnownBits="));
856 // UartSendString(SERIAL_PORT_PC,currentLevelKnownBits, DEC);
857 buffer[1] =
858 0x70; // NVB - Number of Valid Bits: Seven whole uint8_ts
859 // Calculate BCC - Block Check Character
860 buffer[6] = buffer[2] ^ buffer[3] ^ buffer[4] ^ buffer[5];
861 // Calculate CRC_A
862 result = PCD_CalculateCRC(mfrc, buffer, 7, &buffer[7]);
863 if (result != STATUS_OK) {
864 return result;
865 }
866 txLastBits = 0; // 0 => All 8 bits are valid.
867 bufferUsed = 9;
868 // Store response in the last 3 uint8_ts of buffer (BCC and
869 // CRC_A - not needed after tx)
870 responseBuffer = &buffer[6];
871 responseLength = 3;
872 } else { // This is an ANTICOLLISION.
873 // UartSendString(SERIAL_PORT_PC,F("ANTICOLLISION: currentLevelKnownBits="));
874 // UartSendString(SERIAL_PORT_PC,currentLevelKnownBits, DEC);
875 txLastBits = currentLevelKnownBits % 8;
876 count = currentLevelKnownBits /
877 8; // Number of whole uint8_ts in the UID part.
878 index = 2 + count; // Number of whole uint8_ts: SEL + NVB + UIDs
879 buffer[1] =
880 (index << 4) + txLastBits; // NVB - Number of Valid Bits
881 bufferUsed = index + (txLastBits ? 1 : 0);
882 // Store response in the unused part of buffer
883 responseBuffer = &buffer[index];
884 responseLength = sizeof(buffer) - index;
885 }
886
887 // Set bit adjustments
888 rxAlign = txLastBits; // Having a separate variable is overkill. But
889 // it makes the next line easier to read.
890 PCD_WriteRegister(mfrc, BitFramingReg,
891 (rxAlign << 4) +
892 txLastBits); // RxAlign = BitFramingReg[6..4].
893 // TxLastBits =
894 // BitFramingReg[2..0]
895
896 // Transmit the buffer and receive the response.
897 result = PCD_TransceiveData(mfrc, buffer, bufferUsed,
898 responseBuffer, &responseLength,
899 &txLastBits, rxAlign, false);
900 if (result == STATUS_COLLISION) { // More than one PICC in the field
901 // => collision.
902 uint8_t valueOfCollReg = PCD_ReadRegister(
903 mfrc, CollReg); // CollReg[7..0] bits are: ValuesAfterColl
904 // reserved CollPosNotValid CollPos[4:0]
905 if (valueOfCollReg & 0x20) { // CollPosNotValid
906 return STATUS_COLLISION; // Without a valid collision
907 // position we cannot continue
908 }
909 uint8_t collisionPos =
910 valueOfCollReg & 0x1F; // Values 0-31, 0 means bit 32.
911 if (collisionPos == 0) {
912 collisionPos = 32;
913 }
914 if (collisionPos <=
915 currentLevelKnownBits) { // No progress - should not happen
916 return STATUS_INTERNAL_ERROR;
917 }
918 // Choose the PICC with the bit set.
919 currentLevelKnownBits = collisionPos;
920 count = (currentLevelKnownBits - 1) % 8; // The bit to modify
921 index = 1 + (currentLevelKnownBits / 8) +
922 (count ? 1 : 0); // First uint8_t is index 0.
923 buffer[index] |= (1 << count);
924 } else if (result != STATUS_OK) {
925 return result;
926 } else { // STATUS_OK
927 if (currentLevelKnownBits >= 32) { // This was a SELECT.
928 selectDone = true; // No more anticollision
929 // We continue below outside the while.
930 } else { // This was an ANTICOLLISION.
931 // We now have all 32 bits of the UID in this Cascade Level
932 currentLevelKnownBits = 32;
933 // Run loop again to do the SELECT.
934 }
935 }
936 } // End of while (!selectDone)
937
938 // We do not check the CBB - it was constructed by us above.
939
940 // Copy the found UID uint8_ts from buffer[] to uid->uiduint8_t[]
941 index = (buffer[2] == PICC_CMD_CT) ? 3 : 2; // source index in buffer[]
942 uint8_tsToCopy = (buffer[2] == PICC_CMD_CT) ? 3 : 4;
943 for (count = 0; count < uint8_tsToCopy; count++) {
944 uid->uidByte[uidIndex + count] = buffer[index++];
945 }
946
947 // Check response SAK (Select Acknowledge)
948 if (responseLength != 3 ||
949 txLastBits !=
950 0) { // SAK must be exactly 24 bits (1 uint8_t + CRC_A).
951 return STATUS_ERROR;
952 }
953 // Verify CRC_A - do our own calculation and store the control in
954 // buffer[2..3] - those uint8_ts are not needed anymore.
955 result = PCD_CalculateCRC(mfrc, responseBuffer, 1, &buffer[2]);
956 if (result != STATUS_OK) {
957 return result;
958 }
959 if ((buffer[2] != responseBuffer[1]) ||
960 (buffer[3] != responseBuffer[2])) {
961 return STATUS_CRC_WRONG;
962 }
963 if (responseBuffer[0] &
964 0x04) { // Cascade bit set - UID not complete yes
965 cascadeLevel++;
966 } else {
967 uidComplete = true;
968 uid->sak = responseBuffer[0];
969 }
970 } // End of while (!uidComplete)
971
972 // Set correct uid->size
973 uid->size = 3 * cascadeLevel + 1;
974
975 return STATUS_OK;
976} // End PICC_Select()

1381StatusCode MIFARE_SetValue(MFRC522Ptr_t mfrc, uint8_t blockAddr, long value) {
1382 uint8_t buffer[18];
1383
1384 // Translate the long into 4 uint8_ts; repeated 2x in value block
1385 buffer[0] = buffer[8] = (value & 0xFF);
1386 buffer[1] = buffer[9] = (value & 0xFF00) >> 8;
1387 buffer[2] = buffer[10] = (value & 0xFF0000) >> 16;
1388 buffer[3] = buffer[11] = (value & 0xFF000000) >> 24;
1389 // Inverse 4 uint8_ts also found in value block
1390 buffer[4] = ~buffer[0];
1391 buffer[5] = ~buffer[1];
1392 buffer[6] = ~buffer[2];
1393 buffer[7] = ~buffer[3];
1394 // Address 2x with inverse address 2x
1395 buffer[12] = buffer[14] = blockAddr;
1396 buffer[13] = buffer[15] = ~blockAddr;
1397
1398 // Write the whole data block
1399 return MIFARE_Write(mfrc, blockAddr, buffer, 16);
1400} // End MIFARE_SetValue()

1464 bool acceptTimeout
1465 ) {
1466 // acceptTimeout = false;

1495 if (result != STATUS_OK) {

1807 //sectors 32-39)

1809 //sectors 32-39)

1811 // LSB.
