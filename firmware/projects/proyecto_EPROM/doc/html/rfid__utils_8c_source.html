32static int readCardBlock(MFRC522Ptr_t mfrc522, uint8_t sector,
33 uint8_t blockAddr) {
34
35 MIFARE_Key key;
36 int i;
37 // using FFFFFFFFFFFFh which is the default at chip delivery from the
38 // factory
39 for (i = 0; i < 6; i++) {
40 key.keybyte[i] = 0xFF;
41 }
42
43 // Authenticate using key A
44 status = (StatusCode)PCD_Authenticate(mfrc522, PICC_CMD_MF_AUTH_KEY_A,
45 blockAddr, &key, &(mfrc522->uid));
46 if (status != STATUS_OK) {
47 UartSendString(UART_PC,"PCD_Authenticate() failed: ");
48 UartSendString(UART_PC,GetStatusCodeName(status));
49 return -1;
50 }
51
52 // Read data from the block
53 status = (StatusCode)MIFARE_Read(mfrc522, blockAddr, auxBuffer, &size);
54 if (status != STATUS_OK) {
55 UartSendString(UART_PC,"MIFARE_Read() failed: ");
56 UartSendString(UART_PC,GetStatusCodeName(status));
57 return -2;
58 }
59
60 // Halt PICC
61 PICC_HaltA(mfrc522);
62 // Stop encryption on PCD
63 PCD_StopCrypto1(mfrc522);
64
65 return 0;
66}

141int readCardBalance(MFRC522Ptr_t mfrc522) {
142
143 int balance;
144
145 int readStatus = readCardBlock(mfrc522, 1, 4);
146
147 if (readStatus == 0) {
148 // convert the balance bytes to an integer, byte[0] is the MSB
149 balance = (int)auxBuffer[3] | (int)(auxBuffer[2] << 8) |
150 (int)(auxBuffer[1] << 16) | (int)(auxBuffer[0] << 24);
151 } else {
152 balance = -999;
153 }
154
155 return balance;
156}
157
165int writeCardBalance(MFRC522Ptr_t mfrc522, int newBalance) {
166
167 // set the 16 bytes auxBuffer, auxBuffer[0] is the MSB
168 auxBuffer[0] = newBalance >> 24;
169 auxBuffer[1] = newBalance >> 16;
170 auxBuffer[2] = newBalance >> 8;
171 auxBuffer[3] = newBalance & 0x000000FF;
172 int i;
173 for (i = 4; i < size; i++) {
174 auxBuffer[i] = 0xBB;
175 }
176
177 int writeStatus = writeCardBlock(mfrc522, 1, 4);
178
179 return writeStatus;
180}
