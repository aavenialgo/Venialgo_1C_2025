74// Heart Rate Monitor functions takes a sample value and the sample number
75// Returns true if a beat is detected
76// A running average of four samples is recommended for display on the screen.
77bool checkForBeat(int32_t sample)
78{
79 bool beatDetected = false;
80
81 // Save current state
82 IR_AC_Signal_Previous = IR_AC_Signal_Current;
83
84 //This is good to view for debugging
85 //Serial.print("Signal_Current: ");
86 //Serial.println(IR_AC_Signal_Current);
87
88 // Process next data sample
89 IR_Average_Estimated = averageDCEstimator(&ir_avg_reg, sample);
90 IR_AC_Signal_Current = lowPassFIRFilter(sample - IR_Average_Estimated);
91
92 // Detect positive zero crossing (rising edge)
93 if ((IR_AC_Signal_Previous < 0) & (IR_AC_Signal_Current >= 0))
94 {
95
96 IR_AC_Max = IR_AC_Signal_max; //Adjust our AC max and min
97 IR_AC_Min = IR_AC_Signal_min;
98
99 positiveEdge = 1;
100 negativeEdge = 0;
101 IR_AC_Signal_max = 0;
102
103 //if ((IR_AC_Max - IR_AC_Min) > 100 & (IR_AC_Max - IR_AC_Min) < 1000)
104 if (((IR_AC_Max - IR_AC_Min) > 20) & ((IR_AC_Max - IR_AC_Min) < 1000))
105 {
106 //Heart beat!!!
107 beatDetected = true;
108 }
109 }
110
111 // Detect negative zero crossing (falling edge)
112 if ((IR_AC_Signal_Previous > 0) & (IR_AC_Signal_Current <= 0))
113 {
114 positiveEdge = 0;
115 negativeEdge = 1;
116 IR_AC_Signal_min = 0;
117 }
118
119 // Find Maximum value in positive cycle
120 if (positiveEdge & (IR_AC_Signal_Current > IR_AC_Signal_Previous))
121 {
122 IR_AC_Signal_max = IR_AC_Signal_Current;
123 }
124
125 // Find Minimum value in negative cycle
126 if (negativeEdge & (IR_AC_Signal_Current < IR_AC_Signal_Previous))
127 {
128 IR_AC_Signal_min = IR_AC_Signal_Current;
129 }
130
131 return(beatDetected);
132}

134// Average DC Estimator
135int16_t averageDCEstimator(int32_t *p, uint16_t x)
136{
137 *p += ((((long) x << 15) - *p) >> 4);
138 return (*p >> 15);
139}

141// Low Pass FIR Filter
142int16_t lowPassFIRFilter(int16_t din)
143{
144 cbuf[offset] = din;
145
146 int32_t z = mul16(FIRCoeffs[11], cbuf[(offset - 11) & 0x1F]);
147
148 for (uint8_t i = 0 ; i < 11 ; i++)
149 {
150 z += mul16(FIRCoeffs[i], cbuf[(offset - i) & 0x1F] + cbuf[(offset - 22 + i) & 0x1F]);
151 }
152
153 offset++;
154 offset %= 32; //Wrap condition
155
156 return(z >> 15);
157}

159// Integer multiplier
160int32_t mul16(int16_t x, int16_t y)
161{
162 return((long)x * (long)y);
163}
