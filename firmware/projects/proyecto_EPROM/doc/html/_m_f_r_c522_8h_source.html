45enum PCD_Register {
46 // Page 0: Command and status
47 // 0x00 // reserved for future use
48 CommandReg = 0x01 << 1, // starts and stops command execution
49 ComIEnReg = 0x02 << 1, // enable and disable interrupt request control bits
50 DivIEnReg = 0x03 << 1, // enable and disable interrupt request control bits
51 ComIrqReg = 0x04 << 1, // interrupt request bits
52 DivIrqReg = 0x05 << 1, // interrupt request bits
53 ErrorReg = 0x06 << 1, // error bits showing the error status of the last
54 // command executed
55 Status1Reg = 0x07 << 1, // communication status bits
56 Status2Reg = 0x08 << 1, // receiver and transmitter status bits
57 FIFODataReg = 0x09 << 1, // input and output of 64 byte FIFO buffer
58 FIFOLevelReg = 0x0A << 1, // number of bytes stored in the FIFO buffer
59 WaterLevelReg = 0x0B << 1, // level for FIFO underflow and overflow warning
60 ControlReg = 0x0C << 1, // miscellaneous control registers
61 BitFramingReg = 0x0D << 1, // adjustments for bit-oriented frames
62 CollReg = 0x0E << 1, // bit position of the first bit-collision detected on
63 // the RF interface
64 // 0x0F // reserved for future use
65
66 // Page 1: Command
67 // 0x10 // reserved for future use
68 ModeReg = 0x11 << 1, // defines general modes for transmitting and receiving
69 TxModeReg = 0x12 << 1, // defines transmission data rate and framing
70 RxModeReg = 0x13 << 1, // defines reception data rate and framing
71 TxControlReg = 0x14 << 1, // controls the logical behavior of the antenna
72 // driver pins TX1 and TX2
73 TxASKReg = 0x15 << 1, // controls the setting of the transmission modulation
74 TxSelReg = 0x16 << 1, // selects the internal sources for the antenna driver
75 RxSelReg = 0x17 << 1, // selects internal receiver settings
76 RxThresholdReg = 0x18 << 1, // selects thresholds for the bit decoder
77 DemodReg = 0x19 << 1, // defines demodulator settings
78 // 0x1A // reserved for future use
79 // 0x1B // reserved for future use
80 MfTxReg =
81 0x1C << 1, // controls some MIFARE communication transmit parameters
82 MfRxReg =
83 0x1D << 1, // controls some MIFARE communication receive parameters
84 // 0x1E // reserved for future use
85 SerialSpeedReg =
86 0x1F << 1, // selects the speed of the serial UART interface
87
88 // Page 2: Configuration
89 // 0x20 // reserved for future use
90 CRCResultRegH =
91 0x21 << 1, // shows the MSB and LSB values of the CRC calculation
92 CRCResultRegL = 0x22 << 1,
93 // 0x23 // reserved for future use
94 ModWidthReg = 0x24 << 1, // controls the ModWidth setting?
95 // 0x25 // reserved for future use
96 RFCfgReg = 0x26 << 1, // configures the receiver gain
97 GsNReg = 0x27 << 1, // selects the conductance of the antenna driver pins
98 // TX1 and TX2 for modulation
99 CWGsPReg = 0x28 << 1, // defines the conductance of the p-driver output
100 // during periods of no modulation
101 ModGsPReg = 0x29 << 1, // defines the conductance of the p-driver output
102 // during periods of modulation
103 TModeReg = 0x2A << 1, // defines settings for the internal timer
104 TPrescalerReg = 0x2B << 1, // the lower 8 bits of the TPrescaler value. The
105 // 4 high bits are in TModeReg.
106 TReloadRegH = 0x2C << 1, // defines the 16-bit timer reload value
107 TReloadRegL = 0x2D << 1,
108 TCounterValueRegH = 0x2E << 1, // shows the 16-bit timer value
109 TCounterValueRegL = 0x2F << 1,
110
111 // Page 3: Test Registers
112 // 0x30 // reserved for future use
113 TestSel1Reg = 0x31 << 1, // general test signal configuration
114 TestSel2Reg = 0x32 << 1, // general test signal configuration
115 TestPinEnReg = 0x33 << 1, // enables pin output driver on pins D1 to D7
116 TestPinValueReg =
117 0x34
118 << 1, // defines the values for D1 to D7 when it is used as an I/O bus
119 TestBusReg = 0x35 << 1, // shows the status of the internal test bus
120 AutoTestReg = 0x36 << 1, // controls the digital self-test
121 VersionReg = 0x37 << 1, // shows the software version
122 AnalogTestReg = 0x38 << 1, // controls the pins AUX1 and AUX2
123 TestDAC1Reg = 0x39 << 1, // defines the test value for TestDAC1
124 TestDAC2Reg = 0x3A << 1, // defines the test value for TestDAC2
125 TestADCReg = 0x3B << 1 // shows the value of ADC I and Q channels
126 // 0x3C // reserved for production tests
127 // 0x3D // reserved for production tests
128 // 0x3E // reserved for production tests
129 // 0x3F // reserved for production tests
130};

135typedef enum _PCD_Command {
136 PCD_Idle = 0x00, // no action, cancels current command execution
137 PCD_Mem = 0x01, // stores 25 bytes into the internal buffer
138 PCD_GenerateRandomID = 0x02, // generates a 10-byte random ID number
139 PCD_CalcCRC = 0x03, // activates the CRC co-processor or performs a self-test
140 PCD_Transmit = 0x04, // transmits data from the FIFO buffer
141 PCD_NoCmdChange = 0x07, // no command change, can be used to modify the
142 // CommandReg register bits without affecting the
143 // command, for example, the PowerDown bit
144 PCD_Receive = 0x08, // activates the receiver circuits
145 PCD_Transceive = 0x0C, // transmits data from FIFO buffer to antenna and
146 // automatically activates the receiver after
147 // transmission
148 PCD_MFAuthent =
149 0x0E, // performs the MIFARE standard authentication as a reader
150 PCD_SoftReset = 0x0F // resets the MFRC522
151} PCD_Command;

179typedef enum _PICC_Command {
180 // The commands used by the PCD to manage communication with several PICCs
181 // (ISO 14443-3, Type A, section 6.4)
182 PICC_CMD_REQA = 0x26, // REQuest command, Type A. Invites PICCs in state
183 // IDLE to go to READY and prepare for anti collision
184 // or selection. 7 bit frame.
185 PICC_CMD_WUPA = 0x52, // Wake-UP command, Type A. Invites PICCs in state
186 // IDLE and HALT to go to READY(*) and prepare for
187 // anti collision or selection. 7 bit frame.
188 PICC_CMD_CT = 0x88, // Cascade Tag. Not really a command, but used during
189 // anti collision.
190 PICC_CMD_SEL_CL1 = 0x93, // Anti collision/Select, Cascade Level 1
191 PICC_CMD_SEL_CL2 = 0x95, // Anti collision/Select, Cascade Level 2
192 PICC_CMD_SEL_CL3 = 0x97, // Anti collision/Select, Cascade Level 3
193 PICC_CMD_HLTA = 0x50, // HaLT command, Type A. Instructs an ACTIVE PICC to
194 // go to state HALT.
195 // The commands used for MIFARE Classic (from
196 // http://www.mouser.com/ds/2/302/MF1S503x-89574.pdf, Section 9)
197 // Use PCD_MFAuthent to authenticate access to a sector, then use these
198 // commands to read/write/modify the blocks on the sector.
199 // The read/write commands can also be used for MIFARE Ultralight.
200 PICC_CMD_MF_AUTH_KEY_A = 0x60, // Perform authentication with Key A
201 PICC_CMD_MF_AUTH_KEY_B = 0x61, // Perform authentication with Key B
202 PICC_CMD_MF_READ = 0x30, // Reads one 16 byte block from the
203 // authenticated sector of the PICC. Also used for
204 // MIFARE Ultralight.
205 PICC_CMD_MF_WRITE = 0xA0, // Writes one 16 byte block to the
206 // authenticated sector of the PICC. Called
207 // "COMPATIBILITY WRITE" for MIFARE Ultralight.
208 PICC_CMD_MF_DECREMENT = 0xC0, // Decrements the contents of a block and
209 // stores the result in the internal data
210 // register.
211 PICC_CMD_MF_INCREMENT = 0xC1, // Increments the contents of a block and
212 // stores the result in the internal data
213 // register.
214 PICC_CMD_MF_RESTORE =
215 0xC2, // Reads the contents of a block into the internal data register.
216 PICC_CMD_MF_TRANSFER =
217 0xB0, // Writes the contents of the internal data register to a block.
218 // The commands used for MIFARE Ultralight (from
219 // http://www.nxp.com/documents/data_sheet/MF0ICU1.pdf, Section 8.6)
220 // The PICC_CMD_MF_READ and PICC_CMD_MF_WRITE can also be used for MIFARE
221 // Ultralight.
222 PICC_CMD_UL_WRITE = 0xA2 // Writes one 4 byte page to the PICC.
223} PICC_Command;
