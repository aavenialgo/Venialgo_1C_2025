87int8_t convertToBcdArray(uint32_t data, uint8_t n_digits, uint8_t *bcd_number){
88 if (n_digits > 10){
89 return -1;
90 }
91 for (int i = 0; i < n_digits; i++){
92 bcd_number[n_digits-1-i] = data % 10;
93 data /= 10;
94 }
