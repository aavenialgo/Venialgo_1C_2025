68static void controlNivelAguaTask(void *pParameter){
69 float altura_total = 9.5;
70 float altura_actual;
71 float area_base = 31.57;
72
73 uint8_t distancia = 0;
74 while(true){
75 if (medir_agua){
76 distancia = HcSr04ReadDistanceInCentimeters();
77
78 altura_actual = altura_total - (distancia / 100.0f); // Convertir a metros
79 volumen_agua = altura_actual * area_base * 1000; // Convertir a ml
80 if (volumen_agua < 2500)
81 GPIOOn(GPIO_5);

83 GPIOOff(GPIO_5);

92float leerPeso(uint16_t adcValue){
93 //suponiendo ADC 12 bits (0-4095) y de referencia de 3.3 V
94 float voltaje = (adcValue / 4095.0f) * 3.3f;
95 float peso = (voltaje - 0.5f) * (500.0f / (3.3f - 0.5f));
96 if (peso < 0)
97 peso = 0;
98 if (peso > 500)
99 peso = 500;
100 return peso
101}
