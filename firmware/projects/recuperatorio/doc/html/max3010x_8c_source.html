148bool MAX3010X_begin(void) {
149
150 I2C_initialize(400000);
151
152 // Step 1: Initial Communication and Verification
153 // Check that a MAX3010X is connected
154 if (MAX3010X_readPartID() != MAX_30105_EXPECTEDPARTID) {
155 // Error -- Part ID read from MAX3010X does not match expected part ID.
156 // This may mean there is a physical connectivity problem (broken wire, unpowered, etc).
157 return false;
158 }
159
160 // Populate revision ID
161 MAX3010X_readRevisionID();
162
163 return true;
164}

215void MAX3010X_softReset(void) {
216 bitMask(MAX3010X_MODECONFIG, MAX3010X_RESET_MASK, MAX3010X_RESET);
217
218 // Poll for bit to clear, reset is then complete
219 // Timeout after 100ms
220 // unsigned long startTime = millis();
221 unsigned long Time = 0;
222 while (Time < 100)
223 {
224 uint8_t response = readRegister8(MAX3010X_MODECONFIG);
225 if ((response & MAX3010X_RESET) == 0) break; //We're done!
226 DelayMs(1); //Let's not over burden the I2C bus
227 Time++;
228 }
229}

291//Assigning a SLOT_RED_LED will pulse LED
292//Assigning a SLOT_RED_PILOT will ??
293void MAX3010X_enableSlot(uint8_t slotNumber, uint8_t device) {
294
295 uint8_t originalContents;
296
297 switch (slotNumber) {
298 case (1):
299 bitMask(MAX3010X_MULTILEDCONFIG1, MAX3010X_SLOT1_MASK, device);
300 break;
301 case (2):
302 bitMask(MAX3010X_MULTILEDCONFIG1, MAX3010X_SLOT2_MASK, device << 4);
303 break;
304 case (3):
305 bitMask(MAX3010X_MULTILEDCONFIG2, MAX3010X_SLOT3_MASK, device);
306 break;
307 case (4):
308 bitMask(MAX3010X_MULTILEDCONFIG2, MAX3010X_SLOT4_MASK, device << 4);
309 break;
310 default:
311 //Shouldn't be here!
312 break;
313 }
314}

367// Die Temperature
368// Returns temp in C
369float MAX3010X_readTemperature() {
370
371 //DIE_TEMP_RDY interrupt must be enabled
372 //See issue 19: https://github.com/sparkfun/SparkFun_MAX3010x_Sensor_Library/issues/19
373
374 // Step 1: Config die temperature register to take 1 temperature sample
375 writeRegister8(MAX3010X_DIETEMPCONFIG, 0x01);
376
377 // Poll for bit to clear, reading is then complete
378 // Timeout after 100ms
379 unsigned long startTime = 0;
380 while (startTime < 100)
381 {
382 //uint8_t response = readRegister8(_i2caddr, MAX3010X_DIETEMPCONFIG); //Original way
383 //if ((response & 0x01) == 0) break; //We're done!
384
385 //Check to see if DIE_TEMP_RDY interrupt is set
386 uint8_t response = readRegister8(MAX3010X_INTSTAT2);
387 if ((response & MAX3010X_INT_DIE_TEMP_RDY_ENABLE) > 0) break; //We're done!
388 DelayMs(1); //Let's not over burden the I2C bus
389 startTime++;
390 }
391 //TODO How do we want to fail? With what type of error?
392 //? if(millis() - startTime >= 100) return(-999.0);
393
394 // Step 2: Read die temperature register (integer)
395 int8_t tempInt = readRegister8(MAX3010X_DIETEMPINT);
396 uint8_t tempFrac = readRegister8(MAX3010X_DIETEMPFRAC); //Causes the clearing of the DIE_TEMP_RDY interrupt
397
398 // Step 3: Calculate temperature (datasheet pg. 23)
399 return (float)tempInt + ((float)tempFrac * 0.0625);
400}

402// Returns die temp in F
403float MAX3010X_readTemperatureF() {
404 float temp = MAX3010X_readTemperature();
405
406 if (temp != -999.0) temp = temp * 1.8 + 32.0;
407
408 return (temp);
409}

433//Setup the sensor
434//The MAX3010X has many settings. By default we select:
435// Sample Average = 4
436// Mode = MultiLED
437// ADC Range = 16384 (62.5pA per LSB)
438// Sample rate = 50
439//Use the default setup if you are just getting started with the MAX3010X sensor
440void MAX3010X_setup(uint8_t powerLevel, uint8_t sampleAverage, uint8_t ledMode, int sampleRate, int pulseWidth, int adcRange) {
441 MAX3010X_softReset(); //Reset all configuration, threshold, and data registers to POR values
442
443 //FIFO Configuration
444 //-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-
445 //The chip will average multiple samples of same type together if you wish
446 if (sampleAverage == 1) MAX3010X_setFIFOAverage(MAX3010X_SAMPLEAVG_1); //No averaging per FIFO record
447 else if (sampleAverage == 2) MAX3010X_setFIFOAverage(MAX3010X_SAMPLEAVG_2);
448 else if (sampleAverage == 4) MAX3010X_setFIFOAverage(MAX3010X_SAMPLEAVG_4);
449 else if (sampleAverage == 8) MAX3010X_setFIFOAverage(MAX3010X_SAMPLEAVG_8);
450 else if (sampleAverage == 16) MAX3010X_setFIFOAverage(MAX3010X_SAMPLEAVG_16);
451 else if (sampleAverage == 32) MAX3010X_setFIFOAverage(MAX3010X_SAMPLEAVG_32);
452 else MAX3010X_setFIFOAverage(MAX3010X_SAMPLEAVG_4);
453
454 //setFIFOAlmostFull(2); //Set to 30 samples to trigger an 'Almost Full' interrupt
455 MAX3010X_enableFIFORollover(); //Allow FIFO to wrap/roll over
456 //-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-
457
458 //Mode Configuration
459 //-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-
460 if (ledMode == 3) MAX3010X_setLEDMode(MAX3010X_MODE_MULTILED); //Watch all three LED channels
461 else if (ledMode == 2) MAX3010X_setLEDMode(MAX3010X_MODE_REDIRONLY); //Red and IR
462 else MAX3010X_setLEDMode(MAX3010X_MODE_REDONLY); //Red only
463 activeLEDs = ledMode; //Used to control how many uint8_ts to read from FIFO buffer
464 //-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-
465
466 //Particle Sensing Configuration
467 //-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-
468 if(adcRange < 4096) MAX3010X_setADCRange(MAX3010X_ADCRANGE_2048); //7.81pA per LSB
469 else if(adcRange < 8192) MAX3010X_setADCRange(MAX3010X_ADCRANGE_4096); //15.63pA per LSB
470 else if(adcRange < 16384) MAX3010X_setADCRange(MAX3010X_ADCRANGE_8192); //31.25pA per LSB
471 else if(adcRange == 16384) MAX3010X_setADCRange(MAX3010X_ADCRANGE_16384); //62.5pA per LSB
472 else MAX3010X_setADCRange(MAX3010X_ADCRANGE_2048);
473
474 if (sampleRate < 100) MAX3010X_setSampleRate(MAX3010X_SAMPLERATE_50); //Take 50 samples per second
475 else if (sampleRate < 200) MAX3010X_setSampleRate(MAX3010X_SAMPLERATE_100);
476 else if (sampleRate < 400) MAX3010X_setSampleRate(MAX3010X_SAMPLERATE_200);
477 else if (sampleRate < 800) MAX3010X_setSampleRate(MAX3010X_SAMPLERATE_400);
478 else if (sampleRate < 1000) MAX3010X_setSampleRate(MAX3010X_SAMPLERATE_800);
479 else if (sampleRate < 1600) MAX3010X_setSampleRate(MAX3010X_SAMPLERATE_1000);
480 else if (sampleRate < 3200) MAX3010X_setSampleRate(MAX3010X_SAMPLERATE_1600);
481 else if (sampleRate == 3200) MAX3010X_setSampleRate(MAX3010X_SAMPLERATE_3200);
482 else MAX3010X_setSampleRate(MAX3010X_SAMPLERATE_50);
483
484 //The longer the pulse width the longer range of detection you'll have
485 //At 69us and 0.4mA it's about 2 inches
486 //At 411us and 0.4mA it's about 6 inches
487 if (pulseWidth < 118) MAX3010X_setPulseWidth(MAX3010X_PULSEWIDTH_69); //Page 26, Gets us 15 bit resolution
488 else if (pulseWidth < 215) MAX3010X_setPulseWidth(MAX3010X_PULSEWIDTH_118); //16 bit resolution
489 else if (pulseWidth < 411) MAX3010X_setPulseWidth(MAX3010X_PULSEWIDTH_215); //17 bit resolution
490 else if (pulseWidth == 411) MAX3010X_setPulseWidth(MAX3010X_PULSEWIDTH_411); //18 bit resolution
491 else MAX3010X_setPulseWidth(MAX3010X_PULSEWIDTH_69);
492 //-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-
493
494 //LED Pulse Amplitude Configuration
495 //-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-
496 //Default is 0x1F which gets us 6.4mA
497 //powerLevel = 0x02, 0.4mA - Presence detection of ~4 inch
498 //powerLevel = 0x1F, 6.4mA - Presence detection of ~8 inch
499 //powerLevel = 0x7F, 25.4mA - Presence detection of ~8 inch

592//Polls the sensor for new data
593//Call regularly
594//If new data is available, it updates the head and tail in the main struct
595//Returns number of new samples obtained
596uint16_t MAX3010X_check(void)
597{
598 //Read register FIDO_DATA in (3-uint8_t * number of active LED) chunks
599 //Until FIFO_RD_PTR = FIFO_WR_PTR
600
601 uint8_t readPointer = MAX3010X_getReadPointer();
602 uint8_t writePointer = MAX3010X_getWritePointer();
603
604 int numberOfSamples = 0;
605
606 //Do we have new data?
607 if (readPointer != writePointer)
608 {
609 //Calculate the number of readings we need to get from sensor
610 numberOfSamples = writePointer - readPointer;
611 if (numberOfSamples < 0) numberOfSamples += 32; //Wrap condition
612
613 //We now have the number of readings, now calc uint8_ts to read
614 //For this example we are just doing Red and IR (3 uint8_ts each)
615 int uint8_tsLeftToRead = numberOfSamples * activeLEDs * 3;
616
617 //Get ready to read a burst of data from the FIFO register
618 uint8_t transmit_data_buffer;
619 transmit_data_buffer = MAX3010X_FIFODATA;
620 I2C_SelectRegister(MAX30105_ADDRESS, transmit_data_buffer);
621 // _i2cPort->beginTransmission(MAX3010X_ADDRESS);
622 // _i2cPort->write(MAX3010X_FIFODATA);
623 // _i2cPort->endTransmission();
624
625 //We may need to read as many as 288 uint8_ts so we read in blocks no larger than I2C_BUFFER_LENGTH
626 //I2C_BUFFER_LENGTH changes based on the platform. 64 uint8_ts for SAMD21, 32 uint8_ts for Uno.
627 //Wire.requestFrom() is limited to BUFFER_LENGTH which is 32 on the Uno
628 while (uint8_tsLeftToRead > 0)
629 {
630 int toGet = uint8_tsLeftToRead;
631 if (toGet > I2C_BUFFER_LENGTH)
632 {
633 //If toGet is 32 this is bad because we read 6 uint8_ts (Red+IR * 3 = 6) at a time
634 //32 % 6 = 2 left over. We don't want to request 32 uint8_ts, we want to request 30.
635 //32 % 9 (Red+IR+GREEN) = 5 left over. We want to request 27.
636
637 toGet = I2C_BUFFER_LENGTH - (I2C_BUFFER_LENGTH % (activeLEDs * 3)); //Trim toGet to be a multiple of the samples we need to read
638 }
639
640 uint8_tsLeftToRead -= toGet;
641
642 //Request toGet number of uint8_ts from sensor
643 //_i2cPort->requestFrom(MAX3010X_ADDRESS, toGet);
644 uint8_t data_to_read_buffer[32];
645 uint8_t data = 0;
646 //data_to_read_buffer = reg;
647 I2C_requestBytes(MAX30105_ADDRESS, toGet, data_to_read_buffer, 0);
648
649 while (toGet > 0)
650 {
651 sense.head++; //Advance the head of the storage struct
652 sense.head %= STORAGE_SIZE; //Wrap condition
653
654 uint8_t temp[sizeof(uint32_t)]; //Array of 4 uint8_ts that we will convert into long
655 uint32_t tempLong;
656
657 //Burst read three uint8_ts - RED
658 temp[3] = 0;
659 temp[2] = data_to_read_buffer[0];
660 temp[1] = data_to_read_buffer[1];
661 temp[0] = data_to_read_buffer[2];
662
663 //Convert array to long
664 memcpy(&tempLong, temp, sizeof(tempLong));
665
666 tempLong &= 0x3FFFF; //Zero out all but 18 bits
667
668 sense.red[sense.head] = tempLong; //Store this reading into the sense array
669
670 if (activeLEDs > 1)
671 {
672 //Burst read three more uint8_ts - IR
673 temp[3] = 0;
674 temp[2] = data_to_read_buffer[3];
675 temp[1] = data_to_read_buffer[4];
676 temp[0] = data_to_read_buffer[5];
677
678 //Convert array to long
679 memcpy(&tempLong, temp, sizeof(tempLong));
680
681 tempLong &= 0x3FFFF; //Zero out all but 18 bits
682
683 sense.IR[sense.head] = tempLong;
684 }
685
686 if (activeLEDs > 2)
687 {
688 //Burst read three more uint8_ts - Green
689 temp[3] = 0;
690 temp[2] = data_to_read_buffer[6];
691 temp[1] = data_to_read_buffer[7];
692 temp[0] = data_to_read_buffer[8];
693
694 //Convert array to long
695 memcpy(&tempLong, temp, sizeof(tempLong));
696
697 tempLong &= 0x3FFFF; //Zero out all but 18 bits
698
699 sense.green[sense.head] = tempLong;
700 }
701
702 toGet -= activeLEDs * 3;
703 }
704
705 } //End while (uint8_tsLeftToRead > 0)
706
707 } //End readPtr != writePtr
708
709 return (numberOfSamples); //Let the world know how much new data we found
710}

712//Check for new data but give up after a certain amount of time
713//Returns true if new data was found
714//Returns false if new data was not found
715bool MAX3010X_safeCheck(uint8_t maxTimeToCheck)
716{
717 uint32_t markTime = 0;
718
719 while(1)
720 {
721 if(markTime > maxTimeToCheck) return(false);

723 if(MAX3010X_check() == true) //We found new data!
724 return(true);

726 DelayMs(1);
